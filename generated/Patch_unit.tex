\subsubsection{Patch}
\label{unit:Patch}

A rectangular region within a \unitref{Tract}.

\begin{itemize}
  \item SkyMap
  \item Tract
\end{itemize}

\textbf{Value Fields:}
\begin{itemize}
  \item \textbf{patch (int):}
      Unique (with SkyMap and Tract) integer identifier for the Patch.
\end{itemize}
